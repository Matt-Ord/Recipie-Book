\beginrecipie{Tortellini}{10}{20}{6}
\begin{ingredient}
    \begin{main}
        \item \grams{1200} tortellini
    \end{main}
    \begin{subingredient}{Mushroom Sauce}
        \item 6 cloves garlic
        \item 2 onions
        \item \grams{750} mushrooms
        \item 2 cup cream
        \item 1 cup chicken stock
        \item 1/2 cup white wine
        \item 1 cup parmesan
        \item \tablespoons{1 1/2} fresh thyme
    \end{subingredient}
\end{ingredient}
\begin{recipe}
    \step{Fry the onions and garlic in a pan before adding the mushrooms and cooking until golden brown.}
    \step{Add the white wine and sizzle until mostly evapourated. Add broth, cream and parmesan and simmer on a low heat. Add the thyme, salt and pepper to taste.}
    \step{Cook the pasta and combine with the sauce. Serve with salad.}
\end{recipe}
