\beginrecipie{Chicken Parmesan}{10}{20}{6}
\begin{ingredient}
    \begin{main}
        \item 5 chicken breasts
        \item 3 eggs
        \item 1 cup parmesan
        \item \grams{375} mozzarella
        \item broccoli to serve
    \end{main}
    \begin{subingredient}{Breadcrum Mix}
        \item 1/2 cup parmesan
        \item 1 1/2 cup breadcrumbs
        \item 2 cloves garlic
        \item salt
        \item pepper
    \end{subingredient}
    \begin{subingredient}{Ragu}
        \item 4 onions
        \item 6 cloves garlic
        \item 1/3 cup white wine
        \item 3 tins chopped tomato
        \item \tablespoons{1} tomato puree
        \item \tablespoons{1} thyme
        \item \tablespoons{1} basil
        \item \tablespoons{1} oregano
        \item \teaspoons{1} cajun pepper
    \end{subingredient}
    \begin{subingredient}{Potato Rosti}
        \item \grams{2500} potatoes
        \item 1 onion
        \item bacon
        \item paprika
        \item cayenne pepper
        \item salt
    \end{subingredient}
\end{ingredient}
\begin{recipe}
    \step{Slice the chicken into thin fillets and coat in the breadcrumbs before frying.}
    \step{For the ragu fry the onion and garlic in a pan until soft. Add the Add wine, turn up heat and let it simmer rapidly until mostly evaporated. Add the tinned tomatoes along with the spices and simmer.}
    \step{Place chicken in a dish and spoon ragu over the chicken. Sprinkle sauce with mozzarella cheese then parmesan. Drizzle oil over parmesan and bake 15 minutes until cheese is melted.}
    \step{For the potato rostis grate potato into a bowl and use a tea-towel to remove potato water. Combine with finely slice onion, bacon and spices and fry in oil until crispy.}
\end{recipe}
