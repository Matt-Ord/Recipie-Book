\beginrecipie{Spinach \& Ricotta Pasta Bake}{15}{40}{6}
\begin{ingredient}
    \begin{main}
        \item \grams{300} pasta
        \item \cups{1 1/2} mozzarella
    \end{main}
    \begin{subingredient}{ricotta}
        \item \grams{500} ricotta
        \item \grams{50} parmesan
        \item 2 cloves garlic
        \item \grams{350} spinach
        \item \grams{100} mozzarella
    \end{subingredient}
    \begin{subingredient}{pasta sauce}
        \item 2 garlic cloves
        \item \grams{700} passata
        \item \teaspoons{1} garlic powder
        \item \teaspoons{1} onion powder
        \item \teaspoons{2} mixed italian herbs
        \item \teaspoons{1/2} dried chilli flakes
        \item \teaspoons{1} sugar
    \end{subingredient}
\end{ingredient}
\begin{recipe}
    \step{Preheat the oven to 180\degree{}C.}
    \step{Cook pasta for 1 minute less than packet instructions.}
    \step{Mix togeter the pasta sauce ingredients. In a seperate bowl mix the ricotta ingredients.}
    \step{Add pasta to ricotta and stir to combine. Transfer to a heatproof baking dish.}
    \step{Top with Pasta Sauce, top with cheese. Loosely cover with foil.}
    \step{Bake for 25 minutes, then remove the cover and bake for a further 10 minutes or until the cheese is melted and bubbly.}
\end{recipe}
