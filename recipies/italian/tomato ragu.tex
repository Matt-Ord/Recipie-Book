\beginrecipie{Tomato ragu}{10}{30}{4}
\begin{ingredient}
    \begin{main}
        \item 1 onion
        \item 1 clove garlic
        \item \grams{800} chopped tomato
        \item 1 cup fresh basil
    \end{main}
\end{ingredient}
\begin{recipe}
    \step{Fry the garlic and finely chopped onion in a pan.}
    \step{Add tomato and basil. Season with salt and pepper, simmer for 30 minutes.}
\end{recipe}
