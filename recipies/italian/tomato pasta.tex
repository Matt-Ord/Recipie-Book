\beginrecipie{Tomato Pasta}{10}{30}{2}
\begin{ingredient}
    \begin{main}
        \item 1 onion
        \item  \grams{650} baby plum tomatoes
        \item 5 cloves garlic
        \item \grams{30} basil
        \item \grams{300} spaghetti
        \item parmesan to serve
    \end{main}
\end{ingredient}
\begin{recipe}
    \step{Fry the garlic on low before adding the tomatoes whole, followed by the onion. Season with salt and pepper.}
    \step{Add the pasta to a pot of boiling water. Once the tomatoes crack open sqash with a spatula, chop and add the basil.}
    \step{If the sauce begins to look dry add some of the pasta water, and once the pasta is almost cooked transfer into the pan with the sauce.}
    \step{Cook through for a couple of minutes before serving with a decent amount of parmesan.}
\end{recipe}
