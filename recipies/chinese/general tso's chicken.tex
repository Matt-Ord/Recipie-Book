\beginrecipie{General Tso's Chicken}{10}{60}{6}
\begin{ingredient}
    \begin{main}
        \item \kg{1} boneless chicken thighs
        \item \teaspoons{3} ginger
        \item 4 cloves garlic
        \item 2 cups cornstarch
        \item \teaspoons{1} chilli flakes
    \end{main}
    \begin{subingredient}{Marinade}
        \item 2 cloves garlic
        \item \teaspoons{2} ginger
        \item \tablespoons{5} soy sauce
        \item \tablespoons{2} hoisin sauce
        \item \tablespoons{2} rice vinegar
        \item \teaspoons{3} chilli paste
        \item \teaspoons{2} sesame oil
        \item \tablespoons{5} brown sugar
        \item \tablespoons{2} corn flour
        \item 1 cup chicken stock
    \end{subingredient}
\end{ingredient}
\begin{recipe}
    \step{Mix the soy, hoisin, vinegar, chilli and sesame oil. Take out 2 tablespoons of Sauce-Marinade and mix with chicken along with ginger and garlic.}
    \step{Add cornflour into chicken and toss to coat. Add sugar and cornflour to the remaining sauce, mix in the chicken stock.}
    \step{Fry the chicken in sesame oil until golden and crispy.}
    \step{Place chicken on a paper towel lined plate. Wipe pan and fry garlic, ginger and chilli flakes for 30 seconds.}
    \step{Add sauce until it thickens and toss the chicken. Serve with salad and sticky rice.}
\end{recipe}
