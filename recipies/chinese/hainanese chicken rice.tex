\beginrecipie{Hainanese Chicken Rice}{10}{60}{4}
\begin{ingredient}
    \begin{main}
        \item 1 whole chicken
        \item chicken stock
        \item 5--6 slices ginger
        \item 4 bunches spring onion
        \item uncooked jasmine rice
    \end{main}
    \begin{subingredient}{Ginger-Garlic Sauce}
        \item 4-inch piece ginger
        \item 2 cloves garlic
    \end{subingredient}
    \begin{subingredient}{Sweet Soy Sauce}
        \item 3 tablespoons sugar
        \item 1/3 cup soy sauce
    \end{subingredient}
    \begin{subingredient}{Chilli Sauce}
        \item 6 red chiles
        \item 3-inch ginger
        \item 3 cloves garlic
        \item 1 small lime
        \item \teaspoons{1} rice vinegar
        \item \teaspoons{1/2} sesame oil
    \end{subingredient}
\end{ingredient}
\begin{recipe}
    \step{Bring a pot of water along with ginger and spring onions and stock cubes to boil and add the chicken, setting aside the piece of chicken fat at the back cavity.}
    \step{Once the water boils, lift the chicken out of the water to pour out the colder water that is trapped in the cavity. Cover and leave the pot on the stove for 45--50 minutes. To check if the chicken is done stick a toothpick into the drumstick. If the juices run clear, itâ€™s cooked through.}
    \step{Transfer the chicken into cold water / ice for 15 minutes. Drain and leave until ready to serve.}
    \step{To make the rice heat a pan and add the chicken fat. Stir in the minced garlic and fry breifly. Add the uncooked rice and stir continuously for about two minutes. Add to the rice cooker along with chicken stock from the pot and cook.}
    \step{For the ginger-garlic sauce grind the ginger and garlic in a food processor until a paste forms. Gently fry until lightly caramelised. Add salt to taste.}
    \step{To make sweet soy sauce dissolve sugar in 1/3 cup water in a saucepan. Stir constantly until a syrup forms. Combine with the soy sauce and transfer to a dish.}
    \step{For the chilli sauce grind the chilies, ginger, and garlic in a food processor until finely chopped. Combine with the sesame oil, salt, sugar, lime juice, and vinegar. Transfer to a small bowl and add chicken broth 1 tablespoon at a time until a saucy consistency is achieved.}
\end{recipe}
