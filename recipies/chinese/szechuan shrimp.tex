\beginrecipie{Szechuan Shrimp}{10}{25}{6}
\begin{ingredient}
    \begin{main}
        \item \grams{700} prawns
        \item \teaspoons{1} cornstarch
        \item \teaspoons{1} ginger
        \item \teaspoons{3} chilli bean sauce
        \item 6 cloves garlic
        \item \grams{400} shallots
        \item \tablespoons{3} ketchup
        \item \tablespoons{3} hoisin sauce
        \item \tablespoons{3} shaoxing wine
        \item \grams{400} water chestnuts
        \item 2 carrots finely chopped
        \item 2 red bell peppers
        \item 2 cup chicken stock
        \item 4 bunch spring onion.
        \item \tablespoons{3} cornstarch
        \item rice to serve
    \end{main}
\end{ingredient}
\begin{recipe}
    \step{Coat the shrimp in cornstarch and set aside. Fry the ginger, chilli bean sauce, garlic and shallots on low for 2 mins.}
    \step{Stir in the ketchup and hoisin sauce and fry for 2 mins. Add the Shaoxing wine, water chestnuts, carrrots and the red bell pepper and fry for 2 mins before adding the chicken stock.}
    \step{Allow the mixture to simmer before adding the rice vinegar, sesame oil, salt and pepper to taste. Simmer for 3 minutes before adding the prawns.}
    \step{Add the cornfour and allow the suce to thicken. Once the prawns are cooked through serve with rice.}
\end{recipe}
