\beginrecipie{Kung-Pao Chicken}{15}{15}{4}
\begin{ingredient}
    \begin{main}
        \item \grams{400} boneless chicken thighs
        \item \teaspoons{4} sichuan peppercorns
        \item 5 cloves garlic
        \item 4 birdseye chillies
        \item 2 large onions
        \item 2--3 red bell pepper
        \item handful of cashew nuts
    \end{main}
    \begin{subingredient}{Marinade}
        \item \teaspoons{1} sesame oil
        \item \teaspoons{2} sugar
        \item pinch chinese 5 spice
        \item \tablespoons{1} light soy sauce
        \item \tablespoons{1/2} cornstarch
    \end{subingredient}
    \begin{subingredient}{Sauce}
        \item \tablespoons{1} soy sauce
        \item \tablespoons{1} hoisin sauce
        \item \tablespoons{1} rice vinegar
        \item \teaspoons{2} sugar
        \item \teaspoons{1} tomato paste
        \item 1/3 cup chicken stock
    \end{subingredient}
\end{ingredient}
\begin{recipe}
    \step{Soak large red chillies, in hot water for 10--15 minutes.}
    \step{Finely slice the onion and red pepper, and cut the chicken thighs into cubes. Finely dice the garlic and birdseye chillies and crush the sichuan peppercorns.}
    \step{Add all of the marinade ingridents into a bowl and marinade the chicken.  In a separate bowl, mix all of the sauce ingredients together.}
    \step{Heat 1 tablespoon of vegetable oil in a wok to high heat. Add the onions and peppers and stir-fry for 2 minutes.}
    \step{Push the onions and peppers to one side of the wok and add an extra 1/2 a teaspoon of oil to centre of wok. Add the chicken and allow to sear on one side before folding it through with the peppers and onions. Fry until golden brown.}
    \step{Add the chopped garlic, birdseye chillies red chillies and sichuan peppercorns immediately followed by the sauce, and continue to cook for 2 minutes.}
    \step{Add cashew nuts and serve with white rice.}
\end{recipe}
