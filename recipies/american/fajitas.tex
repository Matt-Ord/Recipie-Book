\beginrecipie{Fajitas}{5}{30}{6}
\begin{ingredient}
    \begin{main}
        \item \grams{650} chicken breast
        \item 5 bell peppers
        \item 1 tin black beans
        \item 6 cloves garlic
        \item 3 large onions
        \item 2 sachets fajita seasoning
        \item 12 tortilla wraps
        \item sour cream
        \item rice to serve
    \end{main}
    \begin{subingredient}{Guacamole}
        \item 3 avocados
        \item 1 lime
        \item 1 red onion
        \item \tablespoons{3} chopped cilantro
        \item 2 tomatoes
        \item \teaspoons{1} minced garlic
        \item 1 pinch cayenne pepper
    \end{subingredient}
\end{ingredient}
\begin{recipe}
    \step{In a pan fry the onions and garlic. Slice the chicken into strips and add to the pan.}
    \step{Once the chicken has cooked through add the peppers, black beans and fajita spices.}
    \step{To make the guacamole finely slice the onion and tomatoes then combine with the crushed avacado. Add the cayenne pepper, cilantro and lime juice and mix throughly.}
\end{recipe}
