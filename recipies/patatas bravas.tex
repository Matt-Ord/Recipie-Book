\beginrecipie{Patatas Bravas}{20}{60}{4}
\begin{ingredient}
    \begin{main}
        \item \grams{600} potatoes
        \item \tablespoons{2} olive oil
        \item 2 tins chopped tomato
        \item 3 red chillies
        \item 4 cloves garlic
        \item chopped parsley, to serve
    \end{main}
\end{ingredient}
\begin{recipe}
    \step{Preheat the oven to 225\textdegree{}C.}
    \step{Peel the potatoes and cut into 2.5cm/1in cubes.}
    \step{Heat the olive oil in a large roasting tin until really hot. Place the potatoes in the tin and shake them in the oil until covered. Season well with salt and freshly ground black pepper. Place in the hot oven for about ten minutes until they begin to turn golden-brown.}
    \step{Add the tomatoes, chilli and garlic and stir well. Cook for another 25-35 minutes until the potatoes are soft on the inside and are crisp on the outside.}
    \step{You can either add the parsley and serve, or refrigerate them until the next day, then reheat, adding more oil and then add the parsley before eating.}
\end{recipe}
