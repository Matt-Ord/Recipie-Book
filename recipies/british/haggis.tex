\beginrecipie{Haggis}{30}{60}{6}
\begin{ingredient}
    \begin{main}
        \item \grams{1200} potato
        \item 1 turnip
        \item milk
        \item haggis
        \item sausages
        \item gravy
        \item peas to serve
        \item carrots to serve
    \end{main}
\end{ingredient}
\begin{recipe}
    \step{Peel and roughly chop the potatoes and turnip, boil. Once soft drain the potatoes and mash along with milk, butter, pepper to taste.}
    \step{Cook the haggis and sausages and serve.}
\end{recipe}
