\beginrecipie{Moqueca Baiana}{10}{25}{6}
\begin{ingredient}
    \begin{main}
        \item 750g white fish fillet
        \item 1 1/2 tbsp lime juice
        \item 4 garlic cloves
        \item 2 medium onions
        \item 3 red bell peppers (sliced)
        \item 3 tsp sugar
        \item 2 tbsp cumin
        \item 2 tbsp paprika
        \item 2 tbsp cayenne pepper
        \item 600ml coconut milk
        \item 600ml canned tomatoes
        \item 1 1/2 cup fish broth
        \item 3 tbsp lime juice
        \item rice to serve
    \end{main}
\end{ingredient}
\begin{recipe}
    \step{Combine the fish, lime juice, oil, salt and pepper in a bowl. Cover with cling film and refrigerate for 20 minutes.}
    \step{Heat the 1 tbsp olive oil in a large skillet over high heat. Add the fish and cook until just cooked through and light golden brown. Remove from the skillet and set aside.}
    \step{Reduce the stove to medium high and heat 1 1/2 tbsp olive oil in the same skillet. Add the garlic and onion and cook for 1 1/2 minutes or until the onion is starting to become translucent.}
    \step{Add the bell peppers and cook for 2 minutes.}
    \step{Add the remaining Broth ingredients. Bring to simmer, then turn down to medium. Cook for 15 to 20 minutes or until it thickens. Adjust salt and pepper to taste.}
    \step{Return the fish to the broth to reheat --- about 2 minutes. Stir through lime juice and serve with rice.}
\end{recipe}
