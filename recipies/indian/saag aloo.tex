\beginrecipie{Saag Aloo}{10}{30}{6}
\begin{ingredient}
    \begin{main}
        \item \kg{1} potatoes
        \item \grams{500} spinach
        \item 3 onions
        \item 3 red chillies
        \item \tablespoons{2} grated ginger
        \item \teaspoons{2} black mustard seeds
        \item \teaspoons{2} cumin seeds
        \item \tablespoons{1} turmeric
        \item salt
    \end{main}
\end{ingredient}
\begin{recipe}
    \step{Heat oil in a pan, cook the onions until they begin to soften. Add garlic, chilli, ginger and spices and cook for 1 minute.}
    \step{Add the potato and cook for 10--15 minutes. Add the spinach and stir until it wilts into the mixture.}
\end{recipe}
