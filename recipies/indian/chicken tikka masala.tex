\beginrecipie{Chicken Tikka Masala}{15}{45}{6}
\begin{ingredient}
    \begin{main}
        \item \kg{1} boneless chicken\\ thighs
        \item 1 cup plain yoghurt
        \item 3 onions
        \item \tablespoons{3} grated ginger
        \item 9 cloves garlic
        \item 2 tins chopped tomatoes
        \item \ml{150} cream
        \item \teaspoons{2} sugar
        \item salad to serve
        \item rice to serve
    \end{main}
    \begin{subingredient}{Tikka Marinade}
        \item \tablespoons{1 1/2} fresh ginger
        \item \tablespoons{1 1/2} garam masala
        \item \teaspoons{1 1/2} ground coriander
        \item  \teaspoons{1 1/2} paprika
        \item  \teaspoons{1 1/2} salt
    \end{subingredient}
    \begin{subingredient}{Sauce Spices}
        \item \teaspoons{3} tumeric
        \item \teaspoons{5} garam masala
        \item \teaspoons{3} coriander
        \item \teaspoons{2} cumin
        \item \teaspoons{1/2} cayenne pepper
    \end{subingredient}
\end{ingredient}
\begin{recipe}
    \step{Marinade the chicken before frying on high until the surface of the chicken is charred.}
    \step{Fry the garlic, onions and ginger in butter before adding the sauce spices, frying for 3 mins.}
    \step{Add the chopped tomatoes and simmer for 15 mins before adding the cream, sugar and chicken back to the pan.}
    \step{Once the chicken is cooked through serve with rice and salad.}
\end{recipe}
