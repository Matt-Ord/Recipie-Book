\beginrecipie{Butter Chicken}{10}{60}{6}
\begin{ingredient}
    \begin{main}
        \item \kg{1 1/2} boneless chicken thighs
        \item \tablespoons{4} butter
        \item 2 tins chopped tomatoes
        \item \ml{450} cream
        \item \tablespoons{2} sugar
        \item \teaspoons{2} salt
        \item basmati rice
        \item naan bread
    \end{main}
    \begin{subingredient}{marinade}
        \item \cups{1} plain yoghurt
        \item \teaspoons{2} turmeric
        \item \teaspoons{4} garam masala
        \item \teaspoons{1} cayenne pepper
        \item \teaspoons{2} cumin
        \item \tablespoons{2} ginger
        \item 4 cloves garlic
    \end{subingredient}
\end{ingredient}
\begin{recipe}
    \step{Combine the Marinade ingredients with the chicken in a bowl and allow the mixture to marinade.}
    \step{Heat the butter in a large fry pan. Place chicken in the fry pan and cook for around 3 minutes.}
    \step{Add the tomatoes, cream, sugar and salt. Turn down to low and simmer for 20 minutes.}
    \step{Serve with basmati rice and naan bread.}
\end{recipe}
