\beginrecipie{Tandoori Chicken}{10}{60}{6}
\begin{ingredient}
    \begin{main}
        \item \kg{1.2} boneless chicken\\ thighs
        \item 6 red bell peppers
        \item 2 limes
        \item honey
        \item salad to serve
    \end{main}
    \begin{subingredient}{Marinade}
        \item 8 cloves garlic
        \item \tablespoons{1/2} ground ginger
        \item \teaspoons{1} cayenne pepper
        \item \teaspoons{1} black pepper
        \item \teaspoons{1} turmeric
        \item \teaspoons{1} thyme
        \item \teaspoons{1} biriyani masala
        \item \tablespoons{1} tandoori masala
        \item \teaspoons{1} salt
    \end{subingredient}
    \begin{subingredient}{Saffron Rice}
        \item 8 cloves garlic
        \item \teaspoons{1/4} saffron
        \item 2 cups chicken stock
        \item \teaspoons{1} onion powder
        \item \teaspoons{1/2} turmeric
        \item basmati rice
    \end{subingredient}
\end{ingredient}
\begin{recipe}
    \step{Mix the marinade ingredients in a bowl and rub into the chicken.}
    \step{Slice the peppers and limes into quarters, and place them alongside the chiken in an oven-proof dish.}
    \step{Drizzle with honey and oil, cover dish with foil and place in oven (200Â°C) for just over an hour.}
    \step{Place the saffron in a small bowl of boiling water and allow it to steep. Combine along with the remaining rice ingredients and cook. Serve alongside the chicken and salad.}
\end{recipe}
