\beginrecipie{Lamb \& Macaroni Pasta Bake}{20}{75}{6}
\begin{ingredient}
    \begin{main}
        \item 1 large onion chopped
        \item 2 cloves garlic
        \item 500g lamb mince
        \item 1 tsp ground cinnamon
        \item 1 lamb stock cube
        \item 2 cans chopped tomato
        \item 1 tbsp dried oregano
        \item 400g macaroni
        \item 250g tub ricotta
        \item 50g parmesan
        \item 50ml milk
    \end{main}
\end{ingredient}
\begin{recipe}
    \step{Heat oil in a large pan. Add garlic and onion and cook for 10 minutes.}
    \step{Tip in the lamb mince and brown. Once browned add cinnamon, crumbled stock cube, tomatoes and oregano.}
    \step{Cook for 15 minutes covered and then 15 uncovered until the sauce has thickened.}
    \step{Meanwhile cook the macaroni following pack instructions, pre-heat the oven to 200C.}
    \step{Mix together the ricotta, most of the parmesan and the milk with plenty of seasoning, and stir into the macaroni. Tip the mince into a large baking dish, top with the macaroni mixture, and scatter over the remaining parmesan.}
    \step{Bake for 30 mins until golden and bubbling.}
\end{recipe}
