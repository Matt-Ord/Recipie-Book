\beginrecipie{Chicken \& Mushroom Risotto}{10}{40}{4}
\begin{ingredient}
    \begin{main}
        \item 3 chicken breasts
        \item \grams{250} mushrooms
        \item 1 large onion
        \item 2 bell peppers
        \item 2 cloves garlic
        \item \grams{300} risotto rice
        \item 3 cubes chicken stock
        \item \grams{50} grated parmesan
    \end{main}
\end{ingredient}
\begin{recipe}
    \step{Fry the garlic and finely chopped onion in a pan.}
    \step{Slice the chicken into thin strips and add to the pan. Once cooked dice and add peppers and mushrooms.}
    \step{Add the risotto rice to the pan and slowly add the stock. Continue to top-up with water untill the rice is cooked. Season with pepper.}
    \step{Add the parmesan and stir through before serving.}
\end{recipe}
