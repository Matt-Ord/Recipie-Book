\beginrecipie{Fish Pie}{10}{40}{4}
\begin{ingredient}
    \begin{main}
        \item \grams{1000} white potatoes
        \item 1 large onion
        \item \grams{250} soft cheese
        \item \grams{320} mixed fish
        \item \grams{190} prawns
        \item handful fresh parsley
        \item milk
        \item \ml{100} fish stock
        \item \tablespoons{1} cornflower
    \end{main}
\end{ingredient}
\begin{recipe}
    \step{Preheat oven at 190\textdegree{}C. Put the potatoes in a large pan and cover with water. Bring to the boil and cook for 10--15 minutes until cooked through. Drain and mash, adding a little milk if you wish. Season with salt and pepper.}
    \step{Meanwhile, heat the oil in a large saucepan and gently cook the onion for 5--10 minutes, until softened. Add the soft cheese and the stock and combine well.}
    \step{Blend the cornflour with 2 teaspoons of cold water and stir into the sauce. Cook for a few more minutes until thickened. Add the fish pie mix and cook for 2 minutes before stirring in the prawns and parsley.}
    \step{Pour into an oven dish and top with the mash. Cook for 25--30 minutes until piping hot.}
\end{recipe}
