\beginrecipie{Potato Dauphinoise}{20}{45}{6}
\begin{ingredient}
    \begin{main}
        \item 8 large king Edward potatoes
        \item \ml{500} double cream
        \item \ml{500} milk
        \item \ml{150} vegetable stock
        \item \grams{100} gruyère cheese
    \end{main}
\end{ingredient}
\begin{recipe}
    \step{Preheat the oven to 190°C}
    \step{Tip 500ml double cream, 500ml milk and 3 garlic cloves into a large saucepan and bring to a simmer.}
    \step{Slice 8 large potatoes very finely, about 3--4mm, add them to the cream and simmer for 3 mins until just cooked.}
    \step{Gently stir to separate the potato and stop it sinking and catching on the bottom of the pan.}
    \step{Remove the potatoes with a slotted spoon and place in a wide shallow ovenproof dish so that they are about 5cm in depth.}
    \step{Pour over the garlic infused cream (discarding the garlic) – just enough to seep through the layers and leave a little moisture on the surface.}
    \step{Scatter over 100g grated gruyère cheese, if using, then bake for 30 mins until the potatoes are soft and browned – increase the heat for 5 mins if not brown enough.}
\end{recipe}
