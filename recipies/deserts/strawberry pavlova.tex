\beginrecipie{Strawberry Pavlova}{25}{60}{6}
\begin{ingredient}
    \begin{main}
        \item 4 eff whites
        \item \grams{250} caster sugar
        \item \teaspoons{1} white wine vinegar
        \item \teaspoons{1} cornflour
        \item \teaspoons{1} vanilla extract
    \end{main}
    \begin{subingredient}{topping}
        \item \grams{500} strawberries
        \item \grams{200} redcurrants
        \item \tablespoons{3} icing sugar
    \end{subingredient}
\end{ingredient}
\begin{recipe}
    \step{Preheat the oven to 150Â°C}
    \step{Using a pencil, mark out the circumference of a dinner plate on baking parchment.}
    \step{Whisk 4 egg whites with a hand mixer until they form stiff peaks, then whisk in 250g caster sugar, 1 tbsp at a time, until the meringue looks glossy.}
    \step{Whisk in 1 tsp white wine vinegar, 1 tsp cornflour and 1 tsp vanilla extract.}
    \step{Spread the meringue inside the circle, creating a crater by making the sides a little higher than the middle.}
    \step{Bake for 1 hr, then turn off the heat and let the Pavlova cool completely inside the oven.}
    \step{When the meringue is cool, chop 100g of the hulled strawberries. Mix them with 100g of the redcurrants and 2 tbsp icing sugar.}
    \step{Place in a food processor, blitz until smooth, then push the fruit mixture through a sieve.}
    \step{Whip 350ml double cream with the remaining 1 tbsp icing sugar and spread it over the meringue. Put the remaining 400g hulled and halved strawberries and 100g redcurrants on the cream and finally pour the sauce over the whole lot.}
\end{recipe}
