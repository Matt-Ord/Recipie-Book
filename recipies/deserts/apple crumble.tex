\beginrecipie{Apple Crumble}{15}{40}{4}
\begin{ingredient}
    \begin{main}
        \item \grams{575} bramley apple
        \item \tablespoons{2} caster sugar
    \end{main}
    \begin{subingredient}{Crumble}
        \item \grams{175} plain flour
        \item \grams{110} caster sugar
        \item \grams{110} butter
    \end{subingredient}
    \begin{subingredient}{Toipping}
        \item \tablespoons{1} rolled oats
        \item \tablespoons{1} demerara sugar
    \end{subingredient}
\end{ingredient}
\begin{recipe}
    \step{Preheat the oven to 190C°C}
    \step{Toss 575g peeled, cored and sliced Bramley apples with 2 tbsp golden caster sugar and put in a 23cm round baking dish at least 5cm deep, or a 20cm square dish. Flatten down with your hand to prevent too much crumble falling through.}
    \step{Put 175g plain flour and 110g golden caster sugar in a bowl with a good pinch of salt.}
    \step{Slice in 110g cold butter and rub it in with your fingertips until the mixture looks like moist breadcrumbs. Shake the bowl and any big bits will come to the surface – rub them in.}
    \step{Pour the crumb mix over the apples to form a pile in the centre, then use a fork to even out.}
    \step{Gently press the surface with the back of the fork so the crumble holds together and goes crisp, then lightly drag the fork over the top for a decorative finish.}
    \step{Sprinkle 1 tbsp rolled oats and 1 tbsp demerara sugar over evenly.}
    \step{Set on a baking tray and put in the preheated oven for 35--40 minutes, until the top is golden and the apples feel very soft when you insert a small, sharp knife. Leave to cool for 10 minutes before serving.}
\end{recipe}
