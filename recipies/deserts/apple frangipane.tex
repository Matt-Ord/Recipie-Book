\beginrecipie{Apple Frangipane}{30}{50}{8}
\begin{ingredient}
    \begin{main}
        \item 1/2 brioche loaf
        \item \grams{175} butter
        \item \grams{175} caster sugar
        \item \grams{175} ground almonds
        \item \teaspoons{1} almond extract
        \item 3 eggs
        \item \grams{25} plain flower
        \item 2 red apples
        \item \tablespoons{2} apricot jam
        \item \tablespoons{1} flaked almonds
        \item icing sugar
    \end{main}
\end{ingredient}
\begin{recipe}
    \step{Preheat the oven to 200CÂ°C. Generously grease a shallow ovenproof dish.}
    \step{Thinly slice brioche and arrange in the dish to cover the base, filling in all the gaps, but do not overlap the slices.}
    \step{Put the butter and caster sugar into a food processor and whizz until light and creamy. Add the ground almonds, almond extract, eggs and flour and whizz again until soft and smooth, but do not over-mix.}
    \step{Spoon the mixture over the brioche base and spread to the sides of the dish. Arrange the sliced apples on top in overlapping circles. Bake for about 40 minutes, or until lightly golden all over; it should feel firm in the centre when lightly pressed.}
    \step{Melt the jam with 2 tablespoons of water in a small saucepan. Brush the jam over the pudding and sprinkle with the flaked almonds. Dust with icing sugar and serve warm.}
\end{recipe}
