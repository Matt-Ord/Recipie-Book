\beginrecipie{Sausage \& Spinach Pasta Bake}{10}{60}{5}
\begin{ingredient}
    \begin{main}
        \item 6 pork sausages
        \item \grams{680} passata
        \item \tablespoons{1} oregano
        \item \grams{500} pasta
        \item \grams{400} spinach
        \item \grams{700} cheese sauce
        \item \grams{100} parmesan
        \item 1 egg
    \end{main}
\end{ingredient}
\begin{recipe}
    \step{Heat a large frying pan and brown the sausages, using a wooden spoon to break up the meat. Pour off any fat from the pan, stir in the passata, bring to the boil, then add the oregano and season. Simmer for 10 mins.}
    \step{Cook the pasta for 2 mins less than the pack says, reserve some cooking water, then drain, cool under cold water and drain again. Meanwhile, pour a kettle of boiling water over the spinach in a colander to wilt, then squeeze out all the water.}
    \step{Mix the pasta with the cheese sauce and some of the reserved water, to loosen. Season.}
    \step{In a large dish, add the meat sauce, then the spinach and seasoning. Next, grate a third of the Parmesan, the cheesy pasta and finally the rest of the cheese.}
    \step{Heat oven to 200Â°C. Bake for 40 --- 45 mins, until the dish is piping hot and golden.}
\end{recipe}
