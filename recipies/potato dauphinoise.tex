\beginrecipie{Potato Dauphinoise}{15}{70}{6}
\begin{ingredient}
    \begin{main}
        \item \grams{750} sweet potatoes
        \item \grams{750} king Edward potatoes
        \item \ml{300} double cream
        \item \ml{150} vegetable stock
        \item \grams{50} parmesan cheese
    \end{main}
\end{ingredient}
\begin{recipe}
    \step{Preheat the oven to 220Â°C}
    \step{Peel and slice the sweet potatoes and King Edwards very thinly.}
    \step{Arrange one layer of mixed potatoes in the buttered dish. Season with salt and pepper and drizzle over a little cream and stock.}
    \step{Continue to layer the potatoes and cream and stock until you have  used all the ingredients. Push the potatoes down into the liquid using your hands so they are all covered. Sprinkle with the cheese and cover with foil.}
    \step{Bake in the preheated oven for about 40 minutes. Remove the foil and continue to cook for another 25-30 minutes, or until golden brown and the potatoes are cooked through.}
    \step{Leave to stand for 5 minutes before serving.}
\end{recipe}
