\beginrecipie{Khao Man Gai}{15}{30}{6}
\begin{ingredient}
    \begin{main}
        \item \grams{750} boneless chicken thighs
        \item \tablespoons{8} potato starch
        \item 6 cloves garlic
        \item \tablespoons{2} fish sauce
        \item \tablespoons{1} sugar
        \item salt and pepper
        \item Cucumber, sliced
        \item Coriander
        \item cooked rice
    \end{main}
    \begin{subingredient}{Sauce}
        \item \tablespoons{2} fish sauce
        \item \grams{45} finely sliced ginger
        \item 6 cloves garlic
        \item \grams{60} miso
        \item \tablespoons{3} oyster sauce
    \end{subingredient}
\end{ingredient}
\begin{recipe}
    \step{Fry the garlic in a pan before mixing in the rice. Dissolve the sugar in the fish sauce before adding to the rice.}
    \step{Heat water in a pot and boil the chicken for 15 mins. Combine the sauce ingredients and serve along with the sliced chicken, rice, cucumber and coriander.}
\end{recipe}
