\beginrecipie{Nasi Goreng}{10}{10}{6}
\begin{ingredient}
    \begin{main}
        \item 225g chicken breast
        \item 150g prawns
        \item 4.5 cups cooked jasmine rice
        \item 2 large onions
        \item 2 red chillis
        \item 3 tsp shrimp paste (optional)
        \item 3 bunch spring onion
        \item 1 cucumber
        \item 6 eggs
        \item tomatoes to serve
    \end{main}
    \begin{subingredient}{Kecap Manis}
        \item 6 tbsp soy sauce
        \item 6 tbsp brown sugar
    \end{subingredient}
\end{ingredient}
\begin{recipe}
    \step{Combine soy sauce and brown sugar over medium heat. Bring to a simmer and reduce until it becomes a maple syrup consistency.}
    \step{Heat oil in a large pan over high heat. Add chilli and garlic, stir for 10 seconds.}
    \step{Add onion, cook for 1 minute.}
    \step{Add thinly sliced chicken, cook until it mostly turns white, then add 1 tbsp kecap manis and cook for a further 1 minute or until chicken is mostly cooked through and a bit caramelised.}
    \step{Add rice, 2 tbsp kecap manis, prawns and shrimp paste, if using. Cook, stirring constantly, for 2 minutes until sauce reduces down and rice grains start to caramelise.}
    \step{Take rice off the heat then add cucumber and spring onions, fry eggs to serve}
\end{recipe}
