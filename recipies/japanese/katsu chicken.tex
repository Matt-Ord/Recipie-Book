\beginrecipie{Katsu Chicken}{15}{30}{6}
\begin{ingredient}
    \begin{main}
        \item 2 onions
        \item 2 potatoes
        \item 3 carrots
        \item 3 large chicken breast
        \item 2 cups chicken stock
        \item japanese rice
    \end{main}
    \begin{subingredient}{Curry Roux}
        \item \tablespoons{4} butter
        \item \tablespoons{4} flour
        \item \tablespoons{2} curry powder
        \item \teaspoons{2} garam masala
        \item \teaspoons{1} paprika
        \item \tablespoons{3} soy sauce
        \item \tablespoons{2} mirin
        \item \tablespoons{1} honey
        \item 1 cup chicken stock
    \end{subingredient}
\end{ingredient}
\begin{recipe}
    \step{Melt the butter in a saucepan and add the flour forming a smooth paste. Add the spices followed by the soy sauce, mirin and honey mixing well until the paste reaches a uniform colour.}
    \step{Slowly add the chicken stock to thin out the paste. Roughly chop the vegetables and chicken breast.}
    \step{Fry the onions until soft before adding the vegetables and frying for a furhter 5 mins.}
    \step{Add the roux to the mix along with the remaining chicken stock and boil covered until the potato is cooked through, around 15 mins. Remove the lid and add the chicken, cooking untill the sauce has thickened and the sauce reaches the desired thickness.}
\end{recipe}
