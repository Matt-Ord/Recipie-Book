\beginrecipie{Masoor Dal}{10}{50}{8}
\begin{ingredient}
    \begin{main}
        \item 4 cups red lentils
        \item 4 large onions
        \item 16 cloves garlic
        \item \tablespoons{3} ginger
        \item 6 green chillies
        \item \tablespoons{3} curry powder
        \item \teaspoons{2} mustard seeds
        \item \teaspoons{2} coriander
        \item \teaspoons{1} cumin
        \item \teaspoons{2} salt
        \item 3 tin chopped tomatoes
    \end{main}
    \begin{subingredient}{Biriyani Rice}
        \item 3 onions
        \item biriyani masala
        \item basmati rice
    \end{subingredient}
\end{ingredient}
\begin{recipe}
    \step{Wash the lentils, then add 16 cups water and boil.}
    \step{To prepare the biriyani rice finely dice the onions and fry in oil/ butter until brown. Fry the spices in a little excess oil before stiring into the onions. Add this mix to the rice.}
    \step{Prepare the Dal tadka. Heat oil in a pan and add the garlic, ginger, chillies and some salt. After around 30 second add the onion and fry until soft.}
    \step{Add the curry powder, mustard, corriander, cumin and salt and cook for 30 seconds. Add the tomatoes and allow the mixture to simmer.}
    \step{Skim the lentils as it begins to foam and remove the lid to allow the mixture to simmer until tender.}
    \step{Drain some of the liquid from the lentils before adding the tadka. Allow the mixture to simmer for 5 mins.}
\end{recipe}
