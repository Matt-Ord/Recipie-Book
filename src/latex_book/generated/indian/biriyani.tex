\beginrecipie{Chicken Biriyani}{20}{60}{8}
\begin{ingredient}
    \begin{main}
        \item 750g chicken thighs
        \item 4 large onions
        \item 1 tsp saffron
        \item \cups{1} coriander
    \end{main}
    \begin{subingredient}{Marinade}
        \item 150 ml yoghurt
        \item 125 ml water
        \item 2 tbsp vegetable oil
        \item 6 garlic cloves
        \item 2 tsp finely grated fresh ginger
        \item \tablespoons{4} biriyani masala
        \item 1 tsp salt
    \end{subingredient}
    \begin{subingredient}{Rice}
        \item 2 tbsp salt
        \item 5 cloves
        \item 5 dried bay leaves
        \item 1 star anise
        \item 6 green cardamom pods
        \item 450g basmati rice
    \end{subingredient}
\end{ingredient}
\begin{recipe}
    \step{Mix Marinade in a large pot. Add chicken and coat well. Marinade 20 minutes to overnight.}
    \step{Bring rice to boil in 3/4 of its volume of water, add spices and cook rice until there is no more water. Place saffron in a bowl, leave for 10 minutes.}
    \step{Cook onion until golden brown in lots of oil. Remove onto paper towel lined plate.}
    \step{Place chicken into a pot on medium heat. Cover and cook for 5 minutes.}
    \step{Remove lid. Cook for 5 minutes, turning chicken twice.}
    \step{Scatter over the onion and coriander then top with all the rice. Gently pat down and flatten surface before randomly drizzling saffron across rice surface.}
    \step{Place lid on. Return to stove over medium heat. As soon as you see steam, turn down to low then cook for 25 minutes. Remove from stove, rest with lid on for 10 minutes.}
\end{recipe}
