\beginrecipie{Chicken Adobo}{30}{60}{6}
\begin{ingredient}
    \begin{main}
        \item 18 chicken drums
        \item 3 large onions
        \item 4 green chillies
        \item 12 cloves garlic
        \item \cups{1} white vinegar
        \item \cups{1} soy sauce
        \item \tablespoons{1} whole peppercorns
        \item \teaspoons{1} ground pepper
        \item 8 bay leaves
        \item \tablespoons{1} sugar
        \item peas to serve
        \item rice to serve
    \end{main}
\end{ingredient}
\begin{recipe}
    \step{Fry the chicken wings in the pot until browned, around 8 mins.}
    \step{Remove the chicken wings and fry the garlic, chillies and onions.}
    \step{Once the onions are soft add the remaining ingredients along with the browned chicken to the pot. Cook with the lid on for 35-40 mins until chicken is tender.}
    \step{Remove the chicken and boil with the lid off until the sauce has thickened. Serve with rice and peas.}
\end{recipe}
