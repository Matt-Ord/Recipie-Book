\beginrecipie{Ginataang Manok (Unofficial)}{12}{45}{4}
\begin{ingredient}
    \begin{main}
        \item \grams{900} Chicken Drumsticks
        \item \ml{500} Coconut Milk
        \item Spinach
        \item \tablespoons{2} Garlic
        \item \tablespoons{2} julienned Ginger
        \item 1 Onion
        \item \grams{250} Green Papaya
        \item \teaspoons{1} Paprika
        \item 1 long Green Chilli
        \item 4 thai Red Chilli
        \item Rice
    \end{main}
\end{ingredient}
\begin{recipe}
    \step{Brown the chicken thighs in a large pot. Remove and fry the garlic, onion and ginger until soft.}
    \step{Add back the chicken, paprika and coconut milk and bring to a boil. Simmer for 30 mins.}
    \step{Once the sauce begins to thicken add the chillies and cook for a further 5 mins.}
    \step{Add the green papaya and cook for 5-8 mins before adding the spinach. Season to taste and allow the spinach to wilt. Serve with rice.}
\end{recipe}
