\beginrecipie{Tortellini}{10}{20}{6}
\begin{ingredient}
    \begin{main}
        \item \grams{1200} tortellini
    \end{main}
    \begin{subingredient}{Mushroom Sauce}
        \item 6 cloves garlic
        \item 2 onions
        \item \grams{750} mushrooms
        \item \cups{2} cream
        \item \cups{1} chicken stock
        \item \cups{1} white wine
        \item \cups{1} parmesan
        \item \tablespoons{1.5} fresh thyme
    \end{subingredient}
\end{ingredient}
\begin{recipe}
    \step{Fry the onions and garlic in a pan before adding the mushrooms and cooking until golden brown.}
    \step{Add the white wine and sizzle until mostly evaporated. Add broth, cream and parmesan and simmer on a low heat. Add the thyme, salt and pepper to taste.}
    \step{Cook the pasta and combine with the sauce. Serve with salad.}
\end{recipe}
