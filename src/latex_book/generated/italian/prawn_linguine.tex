\beginrecipie{Prawn Linguine}{10}{20}{6}
\begin{ingredient}
    \begin{main}
        \item \tablespoons{8} butter
        \item \tablespoons{8} garlic
        \item \cups{6} chicken stock
        \item \cups{6} water
        \item \kg{1} linguine
        \item \grams{500} prawns
        \item \cups{1.5} white wine
        \item \cups{0.5} sage
        \item \cups{1} parmesan
        \item \cups{2} cream
        \item \cups{1} parsley
        \item 2 lemons
    \end{main}
    \begin{subingredient}{Bruschetta}
        \item rustic italian bread
        \item 1 clove garlic
        \item 3 tomatoes
        \item 1 small red onion
        \item balsamic vinegar
        \item basil leaves
    \end{subingredient}
\end{ingredient}
\begin{recipe}
    \step{Fry half the garlic in butter in a pot. Add the broth and water and bring to boil before adding the pasta. Once the pasta is cooked lower the heat and simmer until most of the liquid is adsorbed.}
    \step{If using raw prawns fry in butter until cooked through. Add the remaining garlic, sage and white wine to the pan. Once the wine has evaporated remove from heat.}
    \step{Add the shrimp, whipping cream, parmesan and parsley to the pasta. Finish with a squeeze of lemon. Add salt and pepper to taste.}
    \step{For the bruschetta slice and toast the bread. Chop up the tomatoes, red onion basil and combine with the crushed garlic and vinegar.}
\end{recipe}
