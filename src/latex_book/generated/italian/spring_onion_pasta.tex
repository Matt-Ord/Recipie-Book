\beginrecipie{Spring Onion and Pea Spaghetti}{30}{30}{4}
\begin{ingredient}
    \begin{main}
        \item \grams{400} English Peas in pod
        \item 1 Garlic Clove
        \item \grams{350} Spaghetti
        \item \grams{350} Spaghetti
        \item \tablespoons{2} fresh Parsley
        \item Butter
        \item Parmesan Cheese
    \end{main}
\end{ingredient}
\begin{recipe}
    \step{Heat butter in a wide pan. Add spring onion and soften gently, add the peas, salt and 3tbsp water. Simmer until water evaporates.}
    \step{Add garlic, parsley and 3tbsp olive oil. Cover and cook on a low heat for 15 minutes.}
    \step{Add prosciutto and cook until peas are soft, adding oil if more liquid is required.}
    \step{Cook spaghetti, drain and add to peas, serve with parmesan.}
\end{recipe}
