\beginrecipie{Butternut Squash Risotto}{15}{65}{4}
\begin{ingredient}
    \begin{main}
        \item 1 Onion
        \item 2 cloves Garlic
        \item Vegetable Stock
        \item Risotto rice
        \item 1 Butternut Squash
        \item \cups{1} Parmesan Cheese
        \item \cups{0.5} Dry White Wine
        \item Butter
        \item Red Chilli Flakes
        \item Fresh Sage
        \item Kale to serve
    \end{main}
\end{ingredient}
\begin{recipe}
    \step{Roast the butternut squash in the oven until tender. Melt the butter in a frying pan and fry the Onion and garlic until soft.}
    \step{Add the risotto rice to the pan and lightly toast before adding the wine.}
    \step{Allow most of the liquid to boil off before adding the stock slowly, until the rice is tender, stirring constantly.}
    \step{Whilst the rice cooks fry the finely chopped sage until dark green.}
    \step{Once the rice is cooked through add the grated Parmesan, some extra butter and Chilli, stirring vigorously until the rice is thick and creamy}
    \step{Stir through the roasted Butternut squash, leaving a few chunks for garnish. Sprinkle with fried sage and serve along side lightly toasted Kale.}
\end{recipe}
