\beginrecipie{Salmon with Mango Salsa}{30}{25}{4}
\begin{ingredient}
    \begin{main}
        \item 4 salmon fillets
        \item 3 tbsp olive oil
        \item 2 tsp lime zest
        \item 3 tbsp lime juice
        \item 3 garlic cloves
        \item jasmine rice to serve
    \end{main}
    \begin{subingredient}{Salsa}
        \item 1 large mango
        \item 1 red bell pepper
        \item fresh coriander
        \item 1 small red onion
        \item 1 large avocado
        \item 1 tbsp lime juice
    \end{subingredient}
\end{ingredient}
\begin{recipe}
    \step{Mix together olive oil, lime zest, lime juice, garlic and season with salt and pepper to taste.}
    \step{Place salmon in baking dish, cover and allow to marinate.}
    \step{While salmon is marinating prepare rice.}
    \step{In a medium bowl toss together mango, bell pepper, coriander, red onion, avocado, lime juice and olive oil. Season with salt and pepper to taste.}
    \step{Fry salmon in olive oil, about 3 minutes per side or until just cooked through}
\end{recipe}
