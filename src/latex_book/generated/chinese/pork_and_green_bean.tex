\beginrecipie{Pork and Green Bean Stir Fry}{30}{15}{4}
\begin{ingredient}
    \begin{subingredient}{Marinade}
        \item \tablespoons{1} Shaoxing Wine
        \item \tablespoons{1} Light Soy Sauce
        \item \teaspoons{1} Dark Soy Sauce
        \item \tablespoons{2} Ginger
        \item \teaspoons{0.5} Sugar
        \item \teaspoons{1} Cornstarch
    \end{subingredient}
    \begin{main}
        \item \grams{225} Ground Pork
        \item \grams{450} Green Beans
        \item \tablespoons{2} Garlic
        \item 1 Red Pepper
        \item 4 Red Chillies
        \item \teaspoons{0.25} Sugar
        \item \teaspoons{1} Sesame Oil
        \item \tablespoons{1} Light Soy Sauce
        \item \teaspoons{0.25} White Pepper
        \item Spring Onions
        \item Rice
    \end{main}
\end{ingredient}
\begin{recipe}
    \step{Combine the ground meat with the marinade ingredients and allow to rest for 15-20 mins.}
    \step{Heat oil in a wok and fry the chopped green beans until slightly charred.}
    \step{Remove beans from the pan and fry the pork, allowing the mince to brown and crisp.}
    \step{Remove the mince and reduce the heat to medium. Fry the garlic and chillies before adding the red bell peppers.}
    \step{Fry for a minute before adding back in the pork, green beans and spring onions.}
    \step{Add the sugar, sesame oil, soy sauce and pepper along with 2 tablespoons of water. Fry on high for 10-15s and serve with rice.}
\end{recipe}
