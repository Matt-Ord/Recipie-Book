\beginrecipie{Mapo Tofu}{10}{25}{6}
\begin{ingredient}
    \begin{main}
        \item \grams{900} tofu
        \item \grams{500} pork mince
        \item \tablespoons{3} sichuan peppercorns
        \item \tablespoons{6} garlic
        \item \tablespoons{4} chili bean sauce
        \item \tablespoons{2} chilli oil
        \item \cups{1.5} chicken stock
        \item \tablespoons{1} cornflour
        \item \teaspoons{0.5} sesame oil
        \item \teaspoons{0.5} sugar
        \item 4 bunches spring onion
        \item rice to serve
    \end{main}
    \begin{subingredient}{Beansprouts}
        \item \grams{600} beansprouts
        \item 8 cloves garlic
        \item \teaspoons{1} brown sugar
        \item \tablespoons{2} soy sauce
        \item \teaspoons{2} sesame oil
    \end{subingredient}
\end{ingredient}
\begin{recipe}
    \step{Fry the peppercorns in a pan for 30 seconds before adding the garlic and ginger and frying for a further minute.}
    \step{Turn up the heat to high and add the pork mince. Fry until cooked through.}
    \step{Add the chili bean sauce and chicken stock and stir in well. Allow to simmer for a minute or so before adding the cornstarch.}
    \step{After the sauce has thickened add the chilli oil and tofu. Cook for 5 mins before adding the sesame oil, sugar and spring onions.}
    \step{For the beansprouts fry the garlic until it begins to brown, then add the beansprouts and fry for 1 minute. Add the remaining ingredients and fry until cooked through. Serve alongside the tofu and rice.}
\end{recipe}
