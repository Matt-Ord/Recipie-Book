\beginrecipie{Five-Spice Chicken}{20}{40}{4}
\begin{ingredient}
    \begin{main}
        \item 8 chicken thighs
        \item 2 medium white cabbage
        \item 10 cloves garlic
        \item 12 dried red chillies
        \item \tablespoons{1} sichuan peppercorns
        \item 4 bunch spring onions
        \item cornflour
        \item rice to serve
    \end{main}
    \begin{subingredient}{Marinade}
        \item \tablespoons{1} soy sauce
        \item \teaspoons{1} salt
        \item \teaspoons{1} five spice
        \item 1 small red onion
        \item 4 cloves garlic
        \item \teaspoons{1} oil
        \item \tablespoons{1} shaoxing wine
        \item \teaspoons{1} sesame oil
    \end{subingredient}
    \begin{subingredient}{Stir Fry Sauce}
        \item \tablespoons{3} soy sauce
        \item \tablespoons{2} chinese black vinegar
        \item \tablespoons{1} sugar
        \item \tablespoons{2} oyster sauce
    \end{subingredient}
\end{ingredient}
\begin{recipe}
    \step{Mix the marinade ingredients in a bowl and add the chicken. Once marinaded cook the chicken in the oven for 35 mins at 190Â°C.}
    \step{To make the cabbage stir-fry heat oil in a pan on medium and fry the chillies, garlic and sichuan peppercorns. Chop the cabbage into strips and stir fry on high for 2 mins.}
    \step{Mix the sauce ingredients and add to the stir fry along with spring onions. Dissolve 1 tbsp cornstarch in water and add to the pan and fry until the sauce thickens.}
\end{recipe}
