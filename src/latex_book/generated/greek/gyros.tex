\beginrecipie{Gyros}{30}{60}{6}
\begin{ingredient}
    \begin{main}
        \item \grams{750} lamb mince
        \item \grams{150} bacon
        \item 1 onion
        \item 3 cloves garlic
        \item tzatziki
        \item 12 pitta bread
        \item chips to serve
    \end{main}
    \begin{subingredient}{Kebab Spices}
        \item \teaspoons{1} oregano
        \item \teaspoons{2} cumin
        \item \teaspoons{2} coriander
        \item \teaspoons{0.5} cinnamon
        \item \teaspoons{2} salt
    \end{subingredient}
    \begin{subingredient}{Greek Salad}
        \item 6 large vine tomatoes
        \item 1 cucumber
        \item 1 small red onion
        \item olives
        \item \grams{90} feta cheese
        \item olive oil
    \end{subingredient}
\end{ingredient}
\begin{recipe}
    \step{Mix the mince with the spices, and leave to marinade.}
    \step{Add meat, onion, bacon and garlic to a food processor and blend on low until it becomes a paste.}
    \step{Compact meat in a loaf tin, cover with foil and cook in the oven for 20 mins. Remove foil and cook for a furhter 5 mins to brown.}
    \step{To make the salad deeseed the cucumber and tomatoes and cut into irregular wedges. Combine salad ingredients in a bowl.}
\end{recipe}
