\beginrecipie{Pork Bulgogi Rice}{40}{10}{4}
\begin{ingredient}
    \begin{main}
        \item \grams{600} Pork Shoulder
        \item \grams{200} Onion
        \item \grams{30} Spring Onion
        \item Red Chillies
        \item Sesame Seeds
        \item Rice
        \item Cucumber / Lettuce
    \end{main}
    \begin{subingredient}{Marinade}
        \item 1 Royal Gala Apple / pear
        \item 1 Onion
        \item \tablespoons{3} Gochujang
        \item \tablespoons{3} Soy Sauce
        \item \tablespoons{3} Honey
        \item \tablespoons{2} Gochugaru
        \item \tablespoons{2} Mirin
        \item \tablespoons{1} Garlic
        \item \tablespoons{1} Ginger
        \item \teaspoons{0.125} Pepper
    \end{subingredient}
\end{ingredient}
\begin{recipe}
    \step{Blend the marinade ingredients and marinate the meat for at least 30 minutes.}
    \step{In a well heated wok fry the meat on medium high heat until about 3/4 cooked.}
    \step{Add the onion and cook, garnish with chillies, green onion and sesame seeds. Serve with rice and salad.}
\end{recipe}
