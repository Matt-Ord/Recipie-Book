\beginrecipie{Soondubu Jjigae}{15}{5}{4}
\begin{ingredient}
    \begin{main}
        \item 4 inches Kombu
        \item \cups{0.5} dried Anchovies
        \item \cups{2} Kimchi with juice
        \item \grams{250} Pork belly
        \item 1 bunch Spring Onions
        \item 4 cloves Garlic
        \item \tablespoons{2} Gochujang
        \item \tablespoons{1} Soy Sauce
        \item \tablespoons{3} Gochugaru
        \item \grams{700} Tofu
        \item 4 Eggs
        \item Rice
        \item Salad or wilted Spinach to serve
    \end{main}
\end{ingredient}
\begin{recipe}
    \step{Combine kombu and anchovies and cover with water. Bring to a simmer and then remove from heat. Let stand for 5 minutes before straining.}
    \step{Heat oil in a saucepan and add the pork, stirring constantly until cooked through (about 1 minute).}
    \step{Add kimchi, the spring onion whites and garlic and fry until fragrant. Add kimchi juice, gochujang and soy sauce and cook until the vegetables are well coated.}
    \step{Add the broth, gochugaru and tofu and simmer. Remove from heat and stir through the eggs, serving immediately with rice and salad topped with the spring onion greens.}
\end{recipe}
