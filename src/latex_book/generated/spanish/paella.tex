\beginrecipie{Paella}{10}{20}{6}
\begin{ingredient}
    \begin{main}
        \item 1 pack cooked prawns
        \item 1 chorizo
        \item 2 chicken breasts
        \item paella rice
        \item peas
        \item 3 onions
        \item 3 peppers
        \item 2 cups chicken stock
        \item paella spice mix
    \end{main}
\end{ingredient}
\begin{recipe}
    \step{Fry the garlic, chorizo and finely chopped onion in a pan.}
    \step{Slice the chicken into thin strips and add to the pan. Once cooked dice and add peppers, prawns, spice mix and peas.}
    \step{Add the paella rice to the pan and slowly add the stock. Continue to top-up with water until the rice is cooked.}
\end{recipe}
