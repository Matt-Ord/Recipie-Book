\beginrecipie{Nan Ord's Ginger Cake}{15}{45}{5}
\begin{ingredient}
    \begin{main}
        \item \grams{120} stork butter
        \item \grams{120} caster sugar
        \item 2 large eggs
        \item \grams{240} flour
        \item \teaspoons{2} heaped ginger
        \item 1/2 tin golden syrup
    \end{main}
\end{ingredient}
\begin{recipe}
    \step{Preheat the oven to 180Â°C}
    \step{Cream together butter and caster sugar.}
    \step{Combine flour and ground ginger, and mix in with butter as you do.}
    \step{Warm about half a tin of golden syrup by placing tin on a saucepan.}
    \step{Add the golden syrup to the mixture and add a little milk to acieve a soft consistancy.}
    \step{Place mixture in the oven in a well greased and lined dish. Check after 40 minutes --- it may need 10 minutes more}
    \step{Leave to cool, better eaten the next day (if you can wait).}
\end{recipe}
