\beginrecipie{Okonomiyaki}{15}{15}{4}
\begin{ingredient}
    \begin{main}
        \item \grams{150} flour
        \item 9 eggs
        \item \ml{180} chicken stock
        \item \teaspoons{1} salt
        \item \grams{300} white cabbage
        \item 1 onion
        \item 4 bunch spring onions
        \item 4 pork loins
        \item mayonnaise
    \end{main}
    \begin{subingredient}{Sauce}
        \item \tablespoons{4} ketchup
        \item \tablespoons{3} worcestershire sauce
        \item \tablespoons{2} oyster sauce
        \item \tablespoons{1.5} sugar
    \end{subingredient}
\end{ingredient}
\begin{recipe}
    \step{Finely slice the cabbage, onions and slice the pork into thin strips.}
    \step{Whisk together the plain flour, stock, salt and egg together in a bowl. Add onions, spring onion and cabbage to the mix and combine.}
    \step{Fry the pancake on one side for 4 minutes or until brown. Add the pork to the top of the panckake, flip and cook covered for a furhter 5 mins.}
    \step{Combine the okonomiyaki sauce ingredients and pour over the pancakes before serving.}
\end{recipe}
