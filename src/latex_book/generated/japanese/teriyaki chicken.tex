\beginrecipie{Teriyaki Chicken}{15}{30}{6}
\begin{ingredient}
    \begin{main}
        \item \grams{750} boneless chicken\\ thighs
        \item \tablespoons{6} potato starch
        \item salt and pepper
        \item rice to serve
        \item salad to serve
    \end{main}
    \begin{subingredient}{Teriyaki Sauce}
        \item \tablespoons{6} soy sauce
        \item \tablespoons{6} mirin
        \item \tablespoons{6} cooking sake
        \item \tablespoons{3} sugar
    \end{subingredient}
\end{ingredient}
\begin{recipe}
    \step{Put shallow cuts into the chicken to even out the thickness of each slice. Coat the chicken in potato starch, salt and pepper in a ziplock bag removing any excess clumps.}
    \step{Cook the chicken skin side down in oil for 2-3 mins on medium heat until it starts to brown.}
    \step{Flip the chicken, cook for a further 1-2 mins and use kitchen paper to wipe up any excess oil.}
    \step{Add the teriyaki sauce mixture into the pan and cook on medium heat, making sure to flip the chicken to cook evenly. Serve with rice and salad.}
\end{recipe}
