\beginrecipie{Chilli}{10}{20}{4}
\begin{ingredient}
    \begin{main}
        \item \kg{1} mince
        \item 4 onions
        \item 2 red peppers
        \item 6 cloves garlic
        \item 2 cubes beef stock
        \item 2 can chopped tomatoes
        \item 1 can red kidney beans
        \item \tablespoons{4} tomato puree
        \item chilli spice mix
    \end{main}
\end{ingredient}
\begin{recipe}
    \step{Fry the garlic, in a pan before adding the onions. Once softened add the mince and cook until browned before adding the peppers and cooking for a furhter 2 mins.}
    \step{Add the tomatoes, spices and kidney beans before crumbling in the stock cubes and simmer for 10 mins. Serve with rice.}
\end{recipe}
